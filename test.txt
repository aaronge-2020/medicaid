async function getDrugPatentData(drugName) {
  try {
    const response = await fetch(`https://pubchem.ncbi.nlm.nih.gov/rest/pug/compound/name/${drugName}/property/PatentCount,PatentFamilyCount,PatentID/JSON`);
    if (!response.ok) {
      throw new Error('Network response was not ok ' + response.statusText);
    }
    const data = await response.json();
    console.log(data);
  } catch (error) {
    console.error('Error fetching the drug patent data:', error);
  }
}

getDrugPatentData('aspirin');
